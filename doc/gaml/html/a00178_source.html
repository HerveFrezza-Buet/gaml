45  template<typename Iterator>
46  Tabular<Iterator, typename is_secondary_iterator<Iterator>::type> bootstrap(const Iterator& begin, const Iterator& end, unsigned int size) {
47  auto init = [begin,end,size](std::vector<tabular_index_type>& indices) -> void {
48  indices.resize(size);
49  unsigned int range = std::distance(begin,end);
50  for(auto& idx : indices)
51  idx = (tabular_index_type)gaml::random::uniform(0,range);
52  };
53  return Tabular<Iterator, typename is_secondary_iterator<Iterator>::type> (begin,init);
54  }

75  double operator()(const Learner& learner,const DataIterator& begin, const DataIterator& end,
76  const InputOf& inputOf, const OutputOf& outputOf) const {
77  unsigned int size = std::distance(begin,end);
78  unsigned int b;
79 
80  // sets[i] is the ith bootstrapped dataset.
81  std::vector< decltype(gaml::bootstrap(begin,end,size)) > sets;
82 
83  // f[i] is the predictor associated to set i.
84  std::vector<typename Learner::predictor_type> f;
85 
86  // C[i] contains the indices of sets that can be used to test index i.
87  std::map<tabular_index_type, std::set<tabular_index_type>> C;
88 
89 
90  if(verbose)
91  std::cout << "Making " << nb_sets
92  << " bootstrapped sets." << std::endl;
93 
94  auto set_out = std::back_inserter(sets);
95  for(b = 0; b < nb_sets; ++b) *(set_out++) = gaml::bootstrap(begin,end,size);
96 
97  if(verbose) {
98  for(b = 0; b < nb_sets; ++b) {
99  std::cout << " Set " << std::setw(3) << b+1 << '/' << nb_sets << " :";
100 
101  for(auto it_elem = sets[b].begin_index(); it_elem != sets[b].end_index(); ++it_elem)
102  std::cout << ' ' << std::setw(3) << *it_elem;
103  std::cout << std::endl;
104  }
105  }
106 
107 
108  if(verbose)
109  std::cout << "Computing the non-belong-to set list for each example." << std::endl;
110 
111  // We need identity here in order to have the main set as a
112  // tabular collection. Now indexes fit with subset indexes.
113  auto main_set = gaml::identity(begin,end);
114 
115  for(auto it = main_set.begin_index(); it != main_set.end_index(); ++it) {
116  auto i = *it;
117  auto& Ci = C[i];
118  for(b = 0; b < nb_sets; ++b)
119  if(!(sets[b].has(i)))
120  Ci.insert(Ci.begin(),b);
121  // Ci contains the indices of sets that can be used to test i.
122  }
123 
124  if(verbose) {
125  for(auto it = main_set.begin_index(); it != main_set.end_index(); ++it) {
126  auto i = *it;
127  std::cout << " sample " << std::setw(3) << i << " is not in sets {";
128  for(auto elem : C[i])
129  std::cout << ' ' << elem;
130  std::cout << " }" << std::endl;
131  }
132  }
133 
134  if(verbose)
135  std::cout << "Learning on these sets." << std::endl;
136 
137  for(b = 0; b < nb_sets; ++b) {
138  std::ostringstream ostr;
139  if(verbose) {
140  ostr << " Set " << std::setw(3) << b+1 << '/' << nb_sets << " : ";
141  std::cout << ostr.str() << "learning...\r" << std::flush;
142  }
143  auto& Zb = sets[b];
144  f.push_back(learner(Zb.begin(),Zb.end(),inputOf,outputOf));
145  if(verbose)
146  std::cout << ostr.str() << "Done. " << std::endl;
147  }
148 
149  if(verbose)
150  std::cout << "Testing each sample" << std::endl;
151 
152  unsigned int cardD = 0;
153  double sumD = 0;
154 
155  DataIterator zi;
156  auto it = main_set.begin_index();
157  for(zi = begin; it != main_set.end_index(); ++it, ++zi) {
158  auto i = *it;
159  if(verbose)
160  std::cout << " " << std::setw(4) << i+1 << '/' << size << " : ";
161 
162  auto& Ci = C[i];
163 
164  if(Ci.empty()) {
165  if(verbose)
166  std::cout << "belongs to all sets." << std::endl;
167  }
168  else {
169  const typename Learner::predictor_type::input_type& xi = inputOf(*zi);
170  const typename Learner::predictor_type::output_type yi = outputOf(*zi);
171  double sum = 0;
172  for(auto j : Ci)
173  sum += loss(f[j](xi),yi);
174  sum /= Ci.size();
175  if(verbose)
176  std::cout << sum << " (using " << Ci.size() << " samples)" << std::endl;
177  ++cardD;
178  sumD += sum;
179  }
180  }
181 
182  if(cardD == 0)
183  throw gaml::exception::Bootstrap("Every sample belongs to all bootstrapped sets");
184 
185  return sumD/(double)cardD;
186  }

213  const InputOf& inputOf, const OutputOf& outputOf) const {
214  auto predictor = learner(begin,end,inputOf,outputOf);
215  auto empirical = gaml::risk::empirical(loss);

248  template<typename Learner, typename DataIterator, typename InputOf, typename OutputOf>
249  double operator()(const Learner& learner,const DataIterator& begin, const DataIterator& end,
250  const InputOf& inputOf, const OutputOf& outputOf) const {
251  auto predictor = learner(begin,end,inputOf,outputOf);
252  auto empirical = gaml::risk::empirical(loss);
253  double Rn = empirical(predictor,begin,end,inputOf,outputOf);
254  auto estimated = gaml::risk::bootstrap::leave_one_out(loss,nb_sets,false);
255  double R = estimated(learner,begin,end,inputOf,outputOf);
256  unsigned int N = end-begin;
257 
258  if(N == 0)
259  throw gaml::exception::Bootstrap("Empty database provided.");
260 
261 
262  DataIterator i,j;
263 
264  double gamma = 0;
265 
266  for(i = begin; i != end; ++i) {
267  const typename Learner::predictor_type::output_type yi = predictor(inputOf(*i));
268  for(j = begin; j != end; ++j)
269  gamma += loss(outputOf(*j),yi);
270  }
271  gamma /= N*N;
272 
273  double r = (R-Rn)/(gamma-Rn);
274  double omega = 0.632/(1-0.368*r);
275 
276  if(verbose)
277  std::cout << "Empirical risk : Rn : " << Rn << std::endl
278  << "Bootstrapped risk : R : " << R << std::endl
279  << "0-information error rate : gamma : " << gamma << std::endl
280  << "Overfitting rate : r = (R-Rn)/(gamma-Rn) : " << r << std::endl
281  << "Weight : omega = 0.632/(1-0.368*r) : " << omega << std::endl;
282  return (1-omega)*Rn + omega*R;
283  }
284  };

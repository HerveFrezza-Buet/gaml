70  static char expect(std::istream& is, const char* chars,
71  const char* msg = 0) {
72  char c;
73  is >> std::skipws >> c;
74  for (; *chars != 0; ++chars) {
75  if (*chars == -1) {
76  if (is.eof())
77  return -1;
78  } else if (*chars == c)
79  return c;
80  }
81  is.putback(c);
82  std::string fullMessage;
83  if (msg == 0) {
84  fullMessage = "char in \"";
85  (fullMessage += chars) += "\" expected";
86  } else {
87  fullMessage = msg;
88  }
89  throw formatError(is, fullMessage);
90  }

92  static int expect(std::istream& is, const std::vector<const char*>& words) {
93  std::vector<const char*> ptrs = words;
94 
95  char c;
96  is >> c;
97 
98  while (is.good()) {
99  for (std::size_t i = 0; i != ptrs.size(); ++i) {
100  const char*& ptr = ptrs[i];
101  if (ptr != nullptr) {
102  if (c != *ptr)
103  ptr = nullptr;
104  else {
105  ++ptr;
106  if (*ptr == 0)
107  return (int) i;
108  }
109  }
110  }
111  is >> c;
112  }
113  return -1;
114  }

182 template<typename Seq> class SequenceParser: public ParserBase {
183  typedef typename Seq::value_type sequence_value_type;
184  JSONParser<sequence_value_type> elementParser_;
185 public:
186  typedef Seq value_type;
187 
188  void read(std::istream& is, Seq& seq) const {
189  seq.clear();
190  char c = expect(is, "[",
191  "a vector" " starts with" " a left square bracket");
192 
193  is >> std::skipws >> c;
194  if (c != ']') {
195  is.putback(c);
196 
197  std::back_insert_iterator<Seq> ii(seq);
198  sequence_value_type d;
199  while (is.good()) {
200  elementParser_.read(is, d);
201  *ii++ = d;
202  c = expect(is, ",]",
203  "vector" " elements are separated by" " coma");
204  if (c == ']')
205  return;
206  }
207  throw formatError(is,
208  "a vector" " ends with" " a right square bracket");
209  }
210  }
211

213  os << '[';
214  auto it = seq.begin();
215  if (it != seq.end()) {
216  elementParser_.write(os, *it);
217  for (++it; it != seq.end(); ++it) {
218  os << ',';
219  elementParser_.write(os, *it);
220  }
221  os << ']';
222  }

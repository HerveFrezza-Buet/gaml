105  std::pair<output_type,output_type> operator()(const DataIterator& begin, const DataIterator& end,
106  const OutputOf& output_of) const {
107  if(begin == end) throw std::runtime_error("FindTwoClasses : empty dataset");
108 
109  output_type pos_class = output_of(*begin);
110  output_type neg_class = pos_class;
111 
112  auto it = begin;
113  for(++it; it != end; ++it)
114  if((neg_class = output_of(*it)) != pos_class)
115  break;
116  if(it == end) throw std::runtime_error("FindTwoClasses : dataset has only one label");
117 
118  return {pos_class,neg_class};
119  }

413  const InputOf& input_of, const OutputOf& output_of) const {

415  // lambda capture block. We use std::bind to solve this.
416  auto get_output = std::bind(output_of,std::placeholders::_1);
417

419  std::set<typename LEARNER::predictor_type::output_type> labels;
420  for(auto it = begin; it != end; ++it) labels.insert(output_of(*it));
421 
422  // Now, let us learn for each label pair.
423  predictor_type res;
424  for(auto l1 = labels.begin(); l1 != labels.end(); ++l1) {
425  auto l2 = l1;
426  for(++l2; l2 != labels.end(); ++l2) {

429  return get_output(elem) == a || get_output(elem) == b;
430  });
431  res += {algo(dataset.begin(),dataset.end(),input_of,output_of),{*l1,*l2}};
432  }

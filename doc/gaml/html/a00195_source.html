40  template<typename DataIterator, typename Test, typename ValueOf>
41  class RelativeVarianceReduction {
42  public:
43  double operator()(const DataIterator& begin, const DataIterator& end, const Test& test, const ValueOf& value_of) {
44  double n = 0;
45  double mean = 0;
46  double M2 = 0;
47  double var;
48 
49  double n_t = 0;
50  double mean_t = 0;
51  double M2_t = 0;
52  double var_t;
53 
54  double n_f = 0;
55  double mean_f = 0;
56  double M2_f = 0;
57  double var_f;
58 
59  for(auto it = begin; it != end; ++it) {
60  double x = (double)(value_of(*it));
61  double delta = x - mean;
62  mean = mean + delta/(++n);
63  M2 = M2 + delta*(x - mean);
64  if(test(*it)) {
65  double delta_t = x - mean_t;
66  mean_t = mean_t + delta_t/(++n_t);
67  M2_t = M2_t + delta_t*(x - mean_t);
68  }
69  else {
70  double delta_f = x - mean_f;
71  mean_f = mean_f + delta_f/(++n_f);
72  M2_f = M2_f + delta_f*(x - mean_f);
73  }
74  }
75 
76  if(n < 2) return 0; else var = M2/(n-1);
77 
78  if(n_t < 2) var_t = 0; else var_t = M2_t/(n_t-1);
79  if(n_f < 2) var_f = 0; else var_f = M2_f/(n_f-1);
80 
81  return (var - var_t*n_t/n - var_f*n_f/n)/var;
82  }
83  };

114

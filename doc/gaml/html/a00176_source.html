212  auto built_partition = partition.build(begin,end);

226  << " : learning...\r" << std::flush;

230 
231  double risk=0;
232  for(DataIterator it = begin; it != end; ++it) {
233  auto& data = *it;
234  risk += loss(predictor(inputOf(data)),

323  class Variance {
324  public:
325  typedef double output_type;
326  template<typename DataIterator,typename ValueOf>
327  double operator()(const DataIterator& begin, const DataIterator& end, const ValueOf& value_of) const {
328  double n = 0;
329  double mean = 0;
330  double M2 = 0;
331  for(auto it = begin; it != end; ++it) {
332  double x = (double)(value_of(*it));
333  double delta = x - mean;
334  mean = mean + delta/(++n);
335  M2 = M2 + delta*(x - mean);
336  }
337  if (n < 2) return 0;
338  return M2/(n - 1);
339  }

369

372 
373  std::map<VALUE,double,COMP> frequencies;

384  for(auto& kv : frequencies) kv.second /= size;
385  return frequencies;

412  double empirical_risk( gsl_matrix* Phi,
413  gsl_vector* y,
414  std::map<unsigned int, double>& w,
415  std::set<unsigned int>& active_dim) const {
416  int nb_samples = Phi->size1;
417  double risk = 0.0;
418  for(unsigned int i = 0 ; i < nb_samples; ++i) {
419  double ypred = 0.0;
420  for(auto di: active_dim)
421  ypred += gsl_matrix_get(Phi, i, di) * w[di];

423  risk += tmp * tmp;
424  }

196  template<typename DataIterator, typename AttrIterator, typename InputOf, typename OutputOf>
197  bool should_split(const DataIterator& begin, const DataIterator& end,
198  const InputOf& input_of, const OutputOf& output_of,
199  unsigned int nmin,
200  AttrIterator& out_non_constant_attr) {
201  if(std::distance(begin,end) < (typename DataIterator::difference_type)nmin)
202  return false;
203 
204  auto it = begin;
205 
206  const auto& x0 = input_of(*it);
207  auto y0 = output_of(*it);
208 
209  std::vector<bool> is_attr_constant(std::distance(x0.begin(),x0.end()),true);
210  bool is_output_constant = true;
211 
212  auto nb_attr_constant = is_attr_constant.size();
213  for(++it; it != end && (nb_attr_constant > 0 || is_output_constant); ++it) {
214  if(is_output_constant)
215  is_output_constant = output_of(*it) == y0;
216  if(nb_attr_constant > 0) {
217  auto iac = is_attr_constant.begin();
218  const auto& x = input_of(*it);
219  for(auto attr_it = x.begin(); attr_it != x.end(); ++attr_it,++iac)
220  if(*iac) {
221  auto attr_it0 = x0.begin();
222  std::advance(attr_it0,std::distance(is_attr_constant.begin(),iac));
223  if(*attr_it0 != *attr_it) {
224  *iac = false;
225  --nb_attr_constant;
226  }
227  }
228  }
229  }
230 
231  if(is_output_constant || nb_attr_constant == is_attr_constant.size())
232  return false;
233 
234  for(auto iac = is_attr_constant.begin(); iac != is_attr_constant.end(); ++iac)
235  if(!(*iac))
236  *(out_non_constant_attr++) = std::distance(is_attr_constant.begin(),iac);
237  return true;
238  }

267  const InputOf& input_of, const OutputOf& output_of,

271  std::vector<unsigned int> non_constant_attr;

273 
274  if(should_split(begin,end,input_of,output_of,nmin,out_attr)) {
275 
276  unsigned int K = std::min(k,(unsigned int)non_constant_attr.size());
277  std::random_shuffle(non_constant_attr.begin(),non_constant_attr.end());
278  auto attr_begin = non_constant_attr.begin();
279  auto attr_end = attr_begin+K;

281 
282  SCORE<DataIterator,
283  std::function<bool (const typename std::iterator_traits<DataIterator>::value_type&)>,
284  OutputOf> score;
285  ThresholdTest<X> best_test;
286 
287  // Let us consider K random non constant attributes for
288  // splitting, and keep the one with the highest score.
289 
290  double best_score = std::numeric_limits<double>::lowest();
291 
292  for(auto attr_it = attr_begin; attr_it != attr_end; ++attr_it) {
293  auto a = *attr_it;
294 
295  // The following code make a bug when input_of(data) is an
296  // array and when g++ optimization is invoked.
297  //
298  auto value_of = [a,&input_of](const typename DataIterator::value_type& data) -> double {
299  auto input = input_of(data);
300  auto it = input.begin();
301  std::advance(it,a);
302  return *it;
303  };
304 
305  //auto value_of = lambda_value_of<X,const typename DataIterator::value_type>(a,input_of);
306 
307  auto values = gaml::map(begin,end,value_of);
308 
309  auto bounds = std::minmax_element(values.begin(),values.end());
310  double threshold = gaml::random::uniform(*(bounds.first),*(bounds.second));
311  // We need to ensure that threshold > min(bounds) in order
312  // to have non empty collections when splitting the values
313  while(threshold == *(bounds.first))
314  threshold = gaml::random::uniform(*(bounds.first),*(bounds.second));
315 
316  ThresholdTest<X> test(threshold, a);
317 
318  auto score_test = [&test,&input_of](const typename DataIterator::value_type& data) -> bool {
319  return test(input_of(data));
320  };
321 
322  double s = score(begin,end,score_test,output_of);
323  if(s > best_score) {
324  best_score = s;
325  best_test = test;
326  }
327  }

331  auto left = split.true_values;
332  auto right = split.false_values;

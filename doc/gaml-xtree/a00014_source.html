155  template<typename X, typename Y, typename LEAF, typename TEST>
156  std::shared_ptr<Tree<X,Y>> tree_factory(std::istream& is) {
157  char tag,sep;
158  std::shared_ptr<Tree<X,Y>> p;
159  std::shared_ptr<Tree<X,Y>> f;
160  std::shared_ptr<LEAF> l;
161  is >> tag;
162  is.get(sep);
163  switch(tag) {
164  case 'N':
165  p = tree_factory<X,Y,LEAF,TEST>(is);
166  f = tree_factory<X,Y,LEAF,TEST>(is);
167  return std::make_shared<Node<X,Y,TEST>>(is,p,f);
168  break;
169  case 'L':
170  l = std::make_shared<LEAF>();
171  l->read_leaf(is);
172  return l;
173  break;
174  default:
175  std::ostringstream ostr;
176  ostr << "'" << tag << "' leaf/node tag found, 'L' or 'N' expected.";
177  throw std::runtime_error(ostr.str());
178  }

264  template<typename X, typename Y,typename DataIterator, typename InputOf, typename OutputOf>
265  std::shared_ptr<xtree::internal::Tree<X,Y>> build_tree(const DataIterator& begin, const DataIterator& end,
266  const InputOf& input_of, const OutputOf& output_of) {
267  if(std::distance(begin,end) > 10) {
268  auto values = gaml::map(begin,end,output_of);
269  auto bounds = std::minmax_element(values.begin(),values.end());
270 
271  regression::internal::Test<X> test([](const X& x, unsigned int) -> double {return x;});
272  test.threshold = .5*(*(bounds.first)+*(bounds.second));
273 
274  auto filter_function = [input_of,&test](const typename DataIterator::value_type& data) {return test(input_of(data));};
275  auto left = gaml::filter(begin, end, filter_function);
276  auto right = gaml::reject(begin, end, filter_function);
277  auto vleft = gaml::virtualized::sequence(left.begin(), left.end() );
278  auto vright = gaml::virtualized::sequence(right.begin(), right.end());
279  return make_node(test,
280  build_tree<X,Y>(vleft.begin(), vleft.end(), input_of,output_of),
281  build_tree<X,Y>(vright.begin(), vright.end(), input_of,output_of));
282  }
283  else
284  return std::make_shared<regression::internal::Leaf<X>>(gaml::average<double>(begin,end,output_of));
285  }

724  auto it_derivative = begin(derivative);
725  auto it_derivative_end = end(derivative);
726  auto it_tf_derivative = tf_derivative.begin();
727  auto it_input_derivatives = input_derivatives.begin();
728  for(; it_derivative != it_derivative_end; ++it_derivative, ++it_tf_derivative, ++it_input_derivatives)
729  *it_derivative += (*it_tf_derivative) * (*it_input_derivatives);
730  }
731  }
732  else {
733  // We don't have to propagate the call backward
734  // We directly compute the derivative at this level
735  // There is only one of the activities derivatives that is != 0
736  // And we compute the derivative of the form
737  // dj = d/theta z_j
738  // = \sum_k dz_k/dy_k dy_k/dtheta
739  // Where theta is actually a parameter used by one (and only one) of the units
740  // i.e. there is a single dy_k/dtheta which is != 0 !!
741 
742 
743  // Initialize the derivatives to 0
744  std::fill(begin(derivative), end(derivative), 0.0);
745 
746  // We first compute the index of the unit in the previous layer that is involved
747  int parameter_index_in_this_layer = parameter_dim - (_params_end - _nb_params);
748  // we look for the index in the previous layer which is connected via the theta weight
749  // it might be that involved_activity_index_previous == _previous.lsize() in which
750  // case the derivative is actually computed with respect to the bias.
751  int involved_activity_index_previous = parameter_index_in_this_layer % (_previous.lsize() + 1);
752  // and the index of the unit in this layer that is involved, i.e the k where dy_k/dtheta != 0
753  int involved_activity_index = int(parameter_index_in_this_layer / (_previous.lsize() + 1));
754 
755 
756 
757  double dyk_dtheta = 0.0;
758  if(involved_activity_index_previous == _previous._size)
759  // If we derivate according to the bias, dyk/dtheta = dyk/dbk = d/dbk \sum_i w_ki z_i + b_k = 1
760  dyk_dtheta = 1.0; // where k = involved_activity_index
761  else
762  // Otherwise , dyk/dtheta = dyk/dw_kx \sum_i w_ki z_i + b_k = z_x = forward_sweep[...]
763  // where the z_x is the x-th unit in the previous layer
764  dyk_dtheta = *(_previous.begin(forward_sweep) + involved_activity_index_previous);
765 
766 
767  // We compute the derivative of the transfer function only along the k dimension
768  // since the other dimensions bring dy_i / dbk = 0, we don't need them
769  std::vector<double> tf_derivative(_size);
770  for(unsigned int i = 0 ; i < _size; ++i)
771  tf_derivative[i] = compute_output(input, params, i, forward_sweep);
772  _dtf._f(tf_derivative.begin(), tf_derivative.end(), involved_activity_index);
773 
774 
775  auto iter_end = end(derivative);
776  auto iter_tf_derivative = tf_derivative.begin();
777  unsigned int i = 0;
778  for(auto iter = begin(derivative) ; iter != iter_end ; ++iter, ++i, ++iter_tf_derivative) {
779  *iter = (*iter_tf_derivative) * dyk_dtheta;
780  }
781  }

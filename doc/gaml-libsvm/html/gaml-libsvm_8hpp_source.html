126  const InputIterator& begin,
127  const InputIterator& end,
128  const CoefOf& coef_of) {
129  struct svm_node* niter;
130  InputIterator iter = begin;
131  int i;
132 
133  for(niter = res, i=1;
134  iter != end;

358  template<typename DataIterator, typename InputOf, typename OutputOf>
359  Predictor<Input,Output> operator()(const DataIterator& begin,
360  const DataIterator& end,
361  const InputOf& input_of,
362  const OutputOf& label_of) const {
363 
364  // this is manages by a smart pointer at predictor level.
365  struct svm_problem* the_problem = new struct svm_problem;
366 
367  the_problem->l = (int)(std::distance(begin,end));
368  the_problem->y = new double[the_problem->l];
369  the_problem->x = new svm_node*[the_problem->l];
370 
371  svm_node** xiter;
372  double* yiter;
373  DataIterator diter = begin; // no default contructor is required.
374 
375  for(xiter = the_problem->x, yiter = the_problem->y;
376  diter != end;
377  ++diter, ++xiter, ++yiter) {
378  *yiter = label_of(*diter);
379  *xiter = new struct svm_node[nb_nodes_of(input_of(*diter))];
380  nodes_of(input_of(*diter),*xiter);
381  }
382  check(*the_problem,*param);
383  std::shared_ptr<struct svm_problem> problem_ptr(the_problem,internal::free_problem);
384  std::shared_ptr<struct svm_model> model_ptr(train(*the_problem,*param),internal::free_model);
385  return Predictor<Input,Output>(model_ptr,problem_ptr,nb_nodes_of,nodes_of,from_double);
386  }
